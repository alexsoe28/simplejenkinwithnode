pipeline {
	agent any
	tools {nodejs "MyNode"}		
	stages {
		stage("Check Node Version"){
			steps {
			"node --version"
			}
		}
		stage("install dependencies"){
			steps {
			"npm --version"
			"npm install"
			}
		}
		stage("Test"){
			steps {
			"node app.js"
		}
		}
		stage("Release the version"){
			steps {
			echo "Release the Version"
			}
		}
	}
}
